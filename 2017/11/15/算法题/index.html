static isInteger=(num)=>{
return Math.round(num)===num&&Math.ceil(num)===Math.floor(num)
}
static getDivisors=(num)=>{
const ret=[];
if(!isInteger(num)){
return ‘input type error’
}
for(let i=Math.round(num/2);i>1;i–){
if(isInteger(num/i)){
ret.push(num/i)
}
}
if(!ret.length){return ${num} is prime}

}
